30int CBuild::fs::replace(std::string file, std::string token, std::string data) {
31 // Path to tmp file
32 std::string cache;
33 cache = CBUILD_CACHE_DIR + "/" + CBUILD_COPY_CACHE_DIR + "/" +
34 CBuild::fs::path_to_file(file) + ".tmp";
35 // std::cout << cache << "\n";
36 // Open two files
37 CBuild::fs::create({cache}, CBuild::fs::FILE);
38 std::ofstream tmp_file(cache);
39 std::ifstream main_file;
40 main_file.open(file);
41 // Number of replaced tokens
42 static int check_num;
43 check_num = 0;
44 // Read and copy file to temp file line by line
45 std::string line;
46 while (std::getline(main_file, line)) {
47 // If token found - replace it and increment counter
48 while (line.find(token) != std::string::npos) {
49 line.replace(line.find(token), token.size(), data);
50 check_num++;
51 }
52 tmp_file << line << "\n";
53 }
54 // Close files
55 main_file.close();
56 tmp_file.close();
57 // Replace file by it's temp version
58 CBuild::fs::remove(file, true);
59 CBuild::fs::move(cache, file);
60 // Return number of replaced element or -1 if nothing has been replaced
61 if (check_num == 0) return -1;
62 return check_num;
63}

80 std::string line;
81 while (std::getline(main_file, line)) {
82 // We can refer to #define by define
83 if (var_type == "define") var_type = "#define";
84 // If variable found
85 if ((line.find(var_type) != std::string::npos) &&
86 (line.find(var_name) != std::string::npos)) {
87 // All content before variable
88 std::string prefix =
89 line.substr(0, line.find(var_type));
90 // All content after variable
91 std::string postfix =
92 line.substr(line.find(var_name) + var_name.size(),
93 line.size() - 1);
94 // Variable
95 std::string var =
96 var_type + " " + var_name + " = " + data;
97 // Recreate line
98 line = prefix + var + postfix;
99 check_num++;
100 }
101 tmp_file << line << "\n";
102 }
103 // Close files
104 main_file.close();
105 tmp_file.close();
