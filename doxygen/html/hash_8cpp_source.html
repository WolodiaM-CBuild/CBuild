177std::string get_file_content(std::string path) {
178 CBuild::files.clear();
179 CBuild::hash_table_new.clear();
180 CBuild::hash_table_old.clear();
181 std::string ret = "";
182 std::ifstream f(path);
183 std::string line;
184 CBuild::cpp file;
185 file.cpp_path_abs = std::filesystem::absolute(path).string();
186 // Load file and strip comments
187 while (std::getline(f, line)) {
188 line = std::regex_replace(line, CBuild::comment_regex1, "");
189 ret += line;
190 ret += "\n";
191 }
192 f.close();
193 ret = std::regex_replace(ret, CBuild::comment_regex2, "");
194 // Get included files
195 std::smatch match;
196 std::string::const_iterator search_start(ret.cbegin());
197 auto base = std::filesystem::path(path).parent_path();
198 while (std::regex_search(search_start, ret.cend(), match,
199 CBuild::include_search)) {
200 std::string head = match[1];
201 bool err = false;
202 try {
203 head = std::filesystem::canonical(
204 base / std::filesystem::path(head))
205 .string();
206 } catch (const std::exception& e) {
207 err = true;
208 }
209 if (!err) file.hpp_paths_abs.push_back(head);
210 search_start = match.suffix().first;
211 }
212 CBuild::last_file_tmp = file;
213 return ret;
214}
221uint64_t hash(std::string path) {

231 for (auto ch : str) {
