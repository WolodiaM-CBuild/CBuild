146CBuild::types::file_content get_file_data(std::string path,
147 std::string included_from = "") {
148 // Return variable
149 std::string p = path;
150 // Set path to file
151 try {
152 p = CBuild::fs::normalize_path(path, included_from);
153 } catch (const std::exception &e) {
154 }
155 CBuild::types::file_content f;
156 f.path = p;
157 // Load file content to string
158 std::ifstream file(path);
159 std::stringstream buff;
160 buff << file.rdbuf();
161 file.close();
162 std::string str = buff.str();
163 // Strip comments from file
164 str = std::regex_replace(str, CBuild::consts::cxx_line_comment, "");
165 str = std::regex_replace(str, CBuild::consts::c_multiline_comment, "");
166 str = std::regex_replace(str, CBuild::consts::doxygen_comment, "");
167 f.content = str;
168 // Get includes
169 std::smatch match;
170 std::string::const_iterator start(str.cbegin());
171 // Base file path
172 std::string base = CBuild::fs::base(path);
173 while (std::regex_search(start, str.cend(), match,
174 CBuild::consts::include_parser)) {
175 // Get file
176 std::string head = match[1];
177 bool err = false;
178 // Convert to absolute path and check for system includes
179 try {
180 head = CBuild::fs::normalize_path(head, base);
181 } catch (const std::exception &e) {
182 err = true;
183 }
184 // Push path, if it is project include
185 if (!err) {
186 f.includes.push_back(head);
187 }
188 start = match.suffix().first;
189 }
190 return f;
191}
198uint64_t hash(std::string str) {
199 uint64_t hash = 5381;
200 for (auto ch : str) {
201 hash = ((hash << 5) + hash) + ch;

226void procces_files(std::vector<std::string> files, std::string toolchain_id) {
227 CBuild::print_full("\t\t\tCBuild hash v3.0", CBuild::color::MAGENTA);
228 // For every input file
229 for (auto file : files) {
230 // Get file data
231 auto data = CBuild::get_file_data(file);
232 auto cpp = CBuild::types::file(data.path);
233 // Get file hash
234 cpp.hash_new = CBuild::hash(data.content);
235 CBuild::print_full("Calculating hash for \"" + cpp.abs_path + "\"",
236 CBuild::color::GREEN);
237 CBuild::print_full("Need to calculate hash...", CBuild::color::RED);
238 auto old = CBuild::vars::old_hashes.get(cpp.abs_path);
239 if (old != NULL) {
240 cpp.hash_old.set(*old);
241 }
242 // For every invclude
243 for (auto inc : data.includes) {
244 // Get file data
245 auto hdata = CBuild::get_file_data(inc, cpp.abs_path);
246
247 auto hpp = CBuild::types::file(hdata.path);
248 // Check if hash is already stored in hash
249 const uint64_t *hash = NULL;
250 CBuild::print_full("Calculating hash for \"" + hpp.abs_path + "\"",
251 CBuild::color::GREEN);
252 if ((hash = CBuild::vars::headers.get(hpp.abs_path)) != NULL) {
253 // Use stored hash
254 hpp.hash_new = *hash;
255 CBuild::print_full("Using hash from cache.", CBuild::color::GREEN);
256 } else {
257 // Calculate file hash
258 hpp.hash_new = CBuild::hash(hdata.content);
259 CBuild::print_full("Need to calculate hash...", CBuild::color::RED);
260 }
261 auto old = CBuild::vars::old_hashes.get(hpp.abs_path);
262 if (old != NULL) {
263 hpp.hash_old.set(*old);
264 }
265 // Store hash
266 try {
267 CBuild::vars::headers.push_back_check(hpp.abs_path, hpp.hash_new);
268 } catch (std::exception &e) {
269 }
270 // Store proccesed include
271 cpp.includes.push_back(hpp);
272 }
273 // Store proccesed file
274 CBuild::vars::filelist.push_back(cpp);
275 }
