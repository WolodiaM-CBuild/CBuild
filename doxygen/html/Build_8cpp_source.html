428void CBuild::Toolchain::call(std::vector<std::string> *args, bool force, bool debug)
429{
430 CBuild::print("Starting " + this->id + " toolchain in build mode ", CBuild::color::RED);
431 // Save scratch variables
432 this->args = args;
433 this->force = force;
434 // Special link tag for dynamick libraries
435 if (this->build_type == CBuild::DYNAMIC_LIBRARY)
436 this->link_args.push_back("-shared");
437 // Debug mode of compilation
438 if (debug)
439 this->compiler_args.push_back("-g");
440 // Some standart compile and link args
441 this->compiler_args.push_back("-fPIC");
442 this->add_link_arg("-Wl,-z,origin");
443 this->add_compile_arg("-Wl,-z,origin");
444 this->add_link_arg(" -Wl,-rpath,\"\\$ORIGIN\"");
445 this->add_compile_arg(" -Wl,-rpath,\"\\$ORIGIN\"");
446 // For packing in deb fomat
447 if (this->build_type == CBuild::DYNAMIC_LIBRARY)
448 {
449 std::string lib = this->gen_out_name();
450 int pos = lib.find_last_of('/');
451 lib = lib.substr(pos + 1);
452 this->add_link_arg(" -Wl,-soname," + lib);
453 this->add_compile_arg(" -Wl,-soname," + lib);
454 }
455 // Init
456 this->init();
457 CBuild::print("Calling tasks marked as PRE ", CBuild::color::GREEN);
458 // Call all dependencies (tasks) marked as PRE
459 for (unsigned int i = 0; i < this->required.size(); i++)
460 {
461 auto elem = this->required.at(i);
462 if (elem.data == CBuild::PRE)
463 {
464 CBuild::Registry::CallTask(elem.key, {});
465 }
466 }
467 CBuild::print("Calling all required toolchains ", CBuild::color::GREEN);
468 // Call all dependencies (toolchains)
469 for (std::string id : this->depends)
470 {
471 // Call target
472 auto target = CBuild::Registry::GetToolchain(id);
473 if (target != NULL)
474 {
475 target->call(args, force, debug);
476 // get lib name
477 auto out_path = target->gen_out_name();
478 unsigned int end_slash = out_path.find_last_of('/');
479 unsigned int end_dot = out_path.find_last_of('.');
480 std::string out = out_path.substr(end_slash + 4, end_dot - end_slash - 4);
481 // Add lib include
482 this->add_library_include(out);
483 // Add all needed for linking to that library
484 this->add_library_dir(".", CBUILD_BUILD_DIR + "/" + target->get_id() + "/" + CBUILD_BUILD_OUT_DIR + "/");
485 this->add_compile_arg("-Wl,-rpath,\"\\$ORIGIN/../../" + target->get_id() + "/" + CBUILD_BUILD_OUT_DIR + "\"");
486 this->add_link_arg("-Wl,-rpath,\"\\$ORIGIN/../../" + target->get_id() + "/" + CBUILD_BUILD_OUT_DIR + "\"");
487 }
488 }
489 // 5 steps
490 if (!force)
491 CBuild::print("Using precompiled object were possible ", CBuild::color::MAGENTA);
492 CBuild::print("Running pre build tasks ", CBuild::GREEN);
493 this->pre_build();
494 CBuild::print("Running build tasks ", CBuild::GREEN);
495 CBuild::print("Now you can see compiler output", CBuild::MAGENTA);
496 this->build();
497 CBuild::print("Running post build tasks ", CBuild::GREEN);
498 this->post_build();
499 CBuild::print("Running pre link tasks ", CBuild::GREEN);
500 this->pre_link();
501 CBuild::print("Running link tasks ", CBuild::GREEN);
502 CBuild::print("Now you can see linker output", CBuild::MAGENTA);
503 // Special linking for static libraries
504 if (this->build_type == CBuild::STATIC_LIBRARY)
505 {
506 this->link_pack();
507 }
508 else
509 {
510 this->link();
511 }
512 // Last, 6-th step
513 CBuild::print("Running post link tasks ", CBuild::GREEN);
514 this->post_link();
515 CBuild::print("Calling tasks marked as POST ", CBuild::GREEN);
516 // Call all dependencies (tasks) marked as POSt
517 for (unsigned int i = 0; i < this->required.size(); i++)
518 {
519 auto elem = this->required.at(i);
520 if (elem.data == CBuild::POST)
521 {
522 CBuild::Registry::CallTask(elem.key, {});
523 }
524 }
525 CBuild::print("End of execution of toolchain " + this->id + " ", CBuild::RED);
526}
