158void CBuild::Registry::CallTask(std::string name,
159 std::vector<std::string> args) {
160 // Load executed state of task
161 auto check = Registry::task_executed.get_ptr(name);
162 // Check if task exixtst and wasn't executed
163 if (check != NULL && check->data == false) {
164 // Load task
165 auto mod = Registry::tasks.get_ptr(name);
166 // Call all required tasks, recursion ;)
167 for (std::string s : mod->data->self_required()) {
168 Registry::CallTask(s, args);
169 }
170 // Call task
171 mod->data->call(args);
172 // Mark this task es exuted, very easy because of pointers ;)
173 check->data = true;
174 }
175 // Print error if task not found
176 if (check == NULL)
177 printf("Task %s not found. Exiting ...\n", name.c_str());
178}

185CBuild::Toolchain *CBuild::Registry::GetToolchain(std::string name,
186 bool force) {
187 // If we not in force build
188 if (!force) {
189 // Check if toolchain was executed before
190 auto check = Registry::task_executed.get_ptr(name);
191 if (check != NULL && check->data == false) {
192 // Return toolchain pointer if not
193 auto elem = Registry::targets.get_ptr(name);
194 // Mark toolcian as executed
195 check->data = true;
196 return elem->data;
197 }
198 }
199 // We in force build
200 else {
201 // Return toolchain ptr
202 auto elem = Registry::targets.get_ptr(name);
203 if (elem != NULL) return elem->data;
204 }
205 // Some error occured
206 return NULL;
207}
