88void CBuild::Registry::RegistryTask(CBuild::Task *task) {
89 // Save task ptr connected to it's name
90 try {
91 Registry::tasks.push_back_check(task->self_name(), task);
92 } catch (std::exception e) {
93 CBuild::print_full("Error: trying to register task with the same id: " +
94 task->self_name(),
95 CBuild::RED);
96 }
97 // Save task excution state connected to it's name
98 Registry::task_executed.push_back(task->self_name(), false);
99}
100void CBuild::Registry::CallTask(std::string name,
101 std::vector<std::string> args) {
102 // Load executed state of task
103 auto check = Registry::task_executed.get_ptr(name);
104 // Check if task exixtst and wasn't executed
105 if (check != NULL && check->data == false) {
106 // Load task
107 auto mod = Registry::tasks.get_ptr(name);
108 // Call all required tasks, recursion ;)
109 for (std::string s : mod->data->self_required()) {
110 Registry::CallTask(s, args);
111 }
112 // Call task
113 mod->data->call(args);
114 // Mark this task es exuted, very easy because of pointers ;)
115 check->data = true;
116 }
117 // Print error if task not found
118 if (check == NULL)
119 CBuild::print("Task" + name + "not found. Exiting ...", CBuild::RED);
120}

134CBuild::Toolchain *CBuild::Registry::GetToolchain(std::string name,
135 bool force) {
136 // If we not in force build
137 if (!force) {
138 // Check if toolchain was executed before
139 auto check = Registry::task_executed.get_ptr(name);
140 if (check != NULL && check->data == false) {
141 // Return toolchain pointer if not
142 auto elem = Registry::targets.get_ptr(name);
143 // Mark toolcian as executed
144 check->data = true;
145 return elem->data;
146 }
147 }
148 // We in force build
149 else {
150 // Return toolchain ptr
151 auto elem = Registry::targets.get_ptr(name);
152 if (elem != NULL)
153 return elem->data;
154 }
155 // Some error occured
156 return NULL;
157}
