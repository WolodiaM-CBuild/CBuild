123void CBuild::Registry::CallTask(std::string name, std::vector<std::string> args)
124{
125 // Load executed state of task
126 auto check = Registry::task_executed.get_ptr(name);
127 // Check if task exixtst and wasn't executed
128 if (check != NULL && check->data == false)
129 {
130 // Load task
131 auto mod = Registry::tasks.get_ptr(name);
132 // Call all required tasks, recursion ;)
133 for (std::string s : mod->data->self_required())
134 {
135 Registry::CallTask(s, args);
136 }
137 // Call task
138 mod->data->call(args);
139 // Mark this task es exuted, very easy because of pointers ;)
140 check->data = true;
141 }
142 // Print error if task not found
143 if (check == NULL)
144 printf("Task %s not found. Exiting ...\n", name.c_str());
145}

153CBuild::Toolchain *CBuild::Registry::GetToolchain(std::string name, bool force)
154{
155 // If we not in force build
156 if (!force)
157 {
158 // Check if toolchain was executed before
159 auto check = Registry::task_executed.get_ptr(name);
160 if (check != NULL && check->data == false)
161 {
162 // Return toolchain pointer if not
163 auto elem = Registry::targets.get_ptr(name);
164 // Mark toolcian as executed
165 check->data = true;
166 return elem->data;
167 }
168 }
169 // We in force build
170 else
171 {
172 // Return toolchain ptr
173 auto elem = Registry::targets.get_ptr(name);
174 if (elem != NULL) return elem->data;
175 }
176 // Some error occured
177 return NULL;
178}
